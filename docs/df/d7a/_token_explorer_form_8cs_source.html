176  private void CreateView(XLangContext context, TreeNode parent, XLangExpression expr)
177  {
178  TreeNode tn = new TreeNode($"Expr: {expr.Type}");
179  parent.Nodes.Add(tn);
180  if (expr is XLangBinaryOp binOp)
181  {
182  CreateView(context, tn, binOp.Left);
183  CreateView(context, tn, binOp.Right);
184  }
185  else if (expr is XLangUnaryOp unOp)
186  {
187  CreateView(context, tn, unOp.Left);
188  }
189  else if (expr is XLangArrayAccessorOp aacOp)
190  {
191  CreateView(context, tn, aacOp.Left);
192  TreeNode subn = new TreeNode("Params: ");
193  tn.Nodes.Add(subn);
194  foreach (XLangExpression xLangExpression in aacOp.ParameterList)
195  {
196  CreateView(context, subn, xLangExpression);
197  }
198 
199  }
200  else if (expr is XLangInvocationOp invOp)
201  {
202  CreateView(context, tn, invOp.Left);
203  TreeNode subn = new TreeNode("Params: ");
204  tn.Nodes.Add(subn);
205  foreach (XLangExpression xLangExpression in invOp.ParameterList)
206  {
207  CreateView(context, subn, xLangExpression);
